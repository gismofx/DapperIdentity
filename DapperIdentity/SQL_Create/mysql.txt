
CREATE TABLE `identityuser` (
	`ID` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8mb3_general_ci',
	`UserName` VARCHAR(256) NULL DEFAULT NULL COLLATE 'utf8mb3_general_ci',
	`Email` VARCHAR(256) NULL DEFAULT NULL COLLATE 'utf8mb3_general_ci',
	`NormalizedEmail` VARCHAR(256) NULL DEFAULT NULL COLLATE 'utf8mb3_general_ci',
	`NormalizedUserName` VARCHAR(256) NULL DEFAULT NULL COLLATE 'utf8mb3_general_ci',
	`PasswordHash` VARCHAR(256) NULL DEFAULT NULL COLLATE 'utf8mb3_general_ci',
	`EmailConfirmed` BIT(1) NULL DEFAULT NULL,
	`PhoneNumber` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8mb3_general_ci',
	`PhoneNumberConfirmed` BIT(1) NULL DEFAULT NULL,
	`TwoFactorEnabled` BIT(1) NULL DEFAULT NULL,
	`FirstName` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8mb3_general_ci',
	`LastName` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8mb3_general_ci',
	`SecurityStamp` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8mb3_general_ci',
	`IsEnabled` BIT(1) NULL DEFAULT NULL,
	`RefreshToken` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8mb3_general_ci',
	`RefreshTokenExpireTime` DATETIME NULL DEFAULT NULL
)
COLLATE='utf8mb3_general_ci'
ENGINE=InnoDB
;
