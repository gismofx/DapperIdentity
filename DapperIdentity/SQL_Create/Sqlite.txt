CREATE TABLE "IdentityUser" (
	"Id"	TEXT NOT NULL UNIQUE,
	"UserName"	TEXT,
	"Email"	TEXT,
	"NormalizedEmail"	TEXT,
	"NormalizedUserName"	TEXT,
	"PasswordHash"	TEXT,
	"EmailConfirmed"	INT,
	"PhoneNumber"	TEXT,
	"PhoneNumberConfirmed"	INT,
	"TwoFactorEnabled"	INT,
	"FirstName"	TEXT,
	"LastName"	TEXT,
	"SecurityStamp" TEXT,
	PRIMARY KEY("Id")
);

CREATE TABLE "IdentityRole" (
	"Id"	TEXT NOT NULL,
	"Name"	TEXT NOT NULL,
	PRIMARY KEY("Id")
);
